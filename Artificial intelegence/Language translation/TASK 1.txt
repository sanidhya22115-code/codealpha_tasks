class LanguagePair {
    private String pairId;
    private Map<String, String> translationRules;
    
    public LanguagePair(String pairId) {
        this.pairId = pairId;
        this.translationRules = new HashMap<>();
    }
    
    public void addRule(String source, String target) {
        translationRules.put(source.toLowerCase(), target);
    }
    
    public String translate(String text) {
        String[] words = text.toLowerCase().split("\\s+");
        StringBuilder result = new StringBuilder();
        
        for (String word : words) {
            String translation = translationRules.get(word);
            result.append(translation != null ? translation : word)
                  .append(" ");
        }
        
        return result.toString().trim();
    }
}

// SECTION 3: API PROCESSING - Send text to translation engine
class TranslationProcessor {
    private TranslationAPI api;
    
    public TranslationProcessor(TranslationAPI api) {
        this.api = api;
    }
    
    public String processTranslation(String text, String sourceLang, String targetLang) {
        long startTime = System.currentTimeMillis();
        
        // Simulate API call delay
        try {
            Thread.sleep(500);
        } catch (InterruptedException e) {
            Thread.currentThread().interrupt();
        }
        
        String result = api.translateText(text, sourceLang, targetLang);
        long processingTime = System.currentTimeMillis() - startTime;
        
        return String.format("Translation completed in %dms%n%n%s", 
                           processingTime, result);
    }
    
    public boolean isLanguagePairSupported(String source, String target) {
        return api.translateText("test", source, target).contains("‚ùå") == false;
    }
}
